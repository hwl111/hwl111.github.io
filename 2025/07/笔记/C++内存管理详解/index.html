#include <memory>
#include <vector>
#include <chrono>
int main() {
    std::vector<std::shared_ptr<int>> v1, v2;
    auto t1 = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < 100000; ++i) v1.push_back(std::make_shared<int>(i));
    auto t2 = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < 100000; ++i) v2.push_back(std::shared_ptr<int>(new int(i)));
    auto t3 = std::chrono::high_resolution_clock::now();
    std::cout << "make_shared: " << (t2-t1).count() << ", new+shared_ptr: " << (t3-t2).count() << std::endl;
}
